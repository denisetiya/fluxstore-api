API Routing Guide

Login API

Endpoint: /api-login

Method: GET

API Documentation

Endpoint: /api-docs

Method: GET

Register User

Endpoint: /user/register

Required Fields: name, email, password, confirmPassword

Method: POST

Exp body JSON: 

      
        {
          "email": "admin1@gmail.com",
          "password": "qwerty",
          "name": "admin1",
          "confirmPassword": "qwerty"
        }

Login User

Endpoint: /user/login

Required Fields: email, password

Method: POST

Exp body JSON: 

      
        {
          "email": "admin1@gmail.com",
          "password": "qwerty"
        }

Forget Password

Method: POST

Step Route :

Verify Email

Required Fields: email

Endpoint: /user/forgetPassword

Verify Code

Required Fields: email, verifyCode

Endpoint: /user/verifyCode

New Password

Required Fields: email, newPassword

Endpoint: /user/forgetPassword

Product Route

Add Product

Method: POST

required Fields: nameProduct, priceProduct, descProduct(optional), starProduct(optional), categoryProduct, gender(optional), rekomendedProduct(only true, default false), detail : stockProduct, colorProduct, pictureProduct

Endpoint: /product/add

Example body JSON:

      
        {
          "nameProduct" : "tes3234",
          "priceProduct" : "40000",
          "descProduct" : "murah meriah",
          "starProduct" : "5",
          "categoryProduct" : "beju",
          "gender" : "female",
          "rekomendedProduct": true,
          "detail" :[ {
              "stock": "30",
              "color": "black",
              "picture" : "www.googlr.com" 
          },{
              "stock": "10",
              "color": "blue",
              "picture" : "www.googlrad.com" 
          }
          ]
        }

Show All Product

Method: GET

Endpoint: /product/show

Show Single Product

Method: GET

required Fields param: id

Endpoint: /product/show/:id

exp param : http://localhost:3000/product/show/123131kaljdakadj

Delete Product

Method: DELETE

required Fields param: id

Endpoint: /product/delete/:id

exp param : http://localhost:3000/product/delete/123131kaljdakadj

Delete Detail Product

Method: DELETE

required Fields param: id

Endpoint: /product/deleteDetail/:id

exp param : http://localhost:3000/product/deleteDetail/123131kaljdakadj

Update Product

Method: PUT

required Fields param: id

required Fields body: nameProduct, priceProduct, descProduct(optional), starProduct(optional), categoryProduct, gender(optional), rekomendedProduct(only true, default false)

Endpoint: /product/update/:id

exp param : http://localhost:3000/product/updateDetail/123131kaljdakadj

Example body JSON:

      
        {
          "nameProduct": "iphone 12",
          "priceProduct": "25000000",
          "starProduct": "4",
          "categoryProduct": "electronic",
          "gender": "uni",
          "rekomendedProduct": true
        }

Update Detail Product

Method: PUT

required Fields param: id

required Fields body: id(id detail product), stockProduct, colorProduct, pictureProduct

Endpoint: /product/updateDetail/:id

exp param : http://localhost:3000/product/updateDetail/123131kaljdakadj

Example body JSON:

      
        {
          "nameProduct": "product 2",
          "productDetail": [
              {
                  "id": "clv6nj3q400021sqm1wbgg7h5",
                  "stockProduct": "15"
              },
              {
                  "id": "clv6nj3rz00041sqmdvhib823",
                  "stockProduct": "5"
              }
          ]
      }
